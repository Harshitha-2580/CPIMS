    <script>
        let html5QrCode = null;
        let lastScanTime = 0;
        const SCAN_DEBOUNCE_MS = 1500;
        const SCAN_TIMEOUT_MS = 120000;

        document.getElementById('drive-selector').addEventListener('change', function () {
            stopScan();
            loadAttendanceHistory();
        });

        document.getElementById('start-scan').addEventListener('click', startScan);
        document.getElementById('stop-scan').addEventListener('click', stopScan);

        async function loadPlacementDrives() {
            const selector = document.getElementById('drive-selector');
            selector.innerHTML = '<option value="">Loading drives...</option>';

            try {
                const response = await fetch('/api/opportunities/placements/admin/ongoing');
                if (!response.ok) {
                    throw new Error('HTTP ' + response.status + ' while loading drives');
                }
                const data = await response.json();

                if (!data.drives || !Array.isArray(data.drives)) {
                    selector.innerHTML = '<option value="">No drives available</option>';
                    return;
                }

                selector.innerHTML = '<option value="">Select placement drive</option>';
                data.drives.forEach((drive) => {
                    const opt = document.createElement('option');
                    opt.value = drive.id;
                    opt.textContent = `${drive.company_name} / ${drive.job_role} (${drive.event_date ? new Date(drive.event_date).toLocaleDateString() : 'TBA'})`;
                    selector.appendChild(opt);
                });

                if (!selector.value && data.drives.length > 0) {
                    selector.value = data.drives[0].id;
                }

                loadAttendanceHistory();
            } catch (err) {
                console.error('Error loading placement drives:', err);
                selector.innerHTML = '<option value="">Failed to load drives</option>';
                updateScanResult('❌ ' + err.message, 'danger');
            }
        }

        function updateScanResult(message, type = 'info') {
            const scanResult = document.getElementById('scan-result');
            const allowed = {
                info: 'bg-white border border-info text-info',
                success: 'bg-white border border-success text-success',
                warning: 'bg-white border border-warning text-warning',
                danger: 'bg-white border border-danger text-danger'
            };
            scanResult.className = 'status-chip ' + (allowed[type] || allowed.info);
            scanResult.textContent = message;
        }

        async function startScan() {
            const driveId = document.getElementById('drive-selector').value;
            if (!driveId) {
                return alert('Select a drive before scanning');
            }

            updateScanResult('📷 Initializing camera...', 'info');

            if (!html5QrCode) {
                html5QrCode = new Html5Qrcode('qr-reader');
            }

            const config = {
                fps: 12,
                qrbox: { width: 280, height: 280 },
                aspectRatio: 1.0,
                rememberLastUsedCamera: true,
                showTorchButtonIfSupported: true,
                experimentalFeatures: { useBarCodeDetectorIfSupported: true }
            };

            html5QrCode.start(
                { facingMode: { ideal: 'environment' } },
                config,
                (decodedText) => {
                    const now = Date.now();
                    if (now - lastScanTime > SCAN_DEBOUNCE_MS) {
                        lastScanTime = now;
                        updateScanResult('✅ Detected; sending to server...', 'success');
                        processAttendance(decodedText);
                    }
                },
                (errorMessage) => {
                    console.debug('scan error:', errorMessage);
                }
            ).then(() => {
                updateScanResult('📱 Camera started. Position QR code inside box.', 'info');
                document.getElementById('start-scan').style.display = 'none';
                document.getElementById('stop-scan').style.display = 'inline-block';

                setTimeout(() => {
                    if (html5QrCode && html5QrCode.isScanning) {
                        updateScanResult('⏱️ Timeout reached. If no QR detected, move closer or adjust light.', 'warning');
                        stopScan();
                    }
                }, SCAN_TIMEOUT_MS);
            }).catch((err) => {
                let errorText = err.message || 'Failed to start camera';
                if (err.name === 'NotAllowedError' || err.name === 'PermissionDeniedError') {
                    errorText = 'Camera permission denied; allow camera access and retry.';
                } else if (err.name === 'NotFoundError') {
                    errorText = 'No camera device found.';
                } else if (err.name === 'NotReadableError') {
                    errorText = 'Camera in use by another app.';
                }
                updateScanResult('❌ ' + errorText, 'danger');
                document.getElementById('start-scan').style.display = 'inline-block';
                document.getElementById('stop-scan').style.display = 'none';
            });
        }

        function stopScan() {
            if (html5QrCode && html5QrCode.isScanning) {
                html5QrCode.stop().then(() => {
                    updateScanResult('⏹️ Scanner stopped.', 'info');
                    document.getElementById('start-scan').style.display = 'inline-block';
                    document.getElementById('stop-scan').style.display = 'none';
                }).catch(err => console.error('stop error', err));
            }
        }

        async function processAttendance(qrData) {
            try {
                const driveId = document.getElementById('drive-selector').value;
                const response = await fetch('/api/attendance/scan', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ qrData, opportunityId: driveId })
                });

                const result = await response.json();
                const resultArea = document.getElementById('qr-result');
                if (response.ok) {
                    resultArea.className = 'alert alert-success';
                    resultArea.textContent = `✅ ${result.message || 'Attendance successfully recorded.'}`;
                } else {
                    resultArea.className = 'alert alert-warning';
                    resultArea.textContent = `⚠️ ${result.message || result.error || 'Unable to record attendance.'}`;
                }
                resultArea.style.display = 'block';
                setTimeout(() => { resultArea.style.display = 'none'; }, 4500);
                loadAttendanceHistory();
            } catch (err) {
                console.error('processAttendance error', err);
                const resultArea = document.getElementById('qr-result');
                resultArea.className = 'alert alert-danger';
                resultArea.textContent = '❌ Error sending data: ' + err.message;
                resultArea.style.display = 'block';
            }
        }

        async function loadAttendanceHistory() {
            const driveId = document.getElementById('drive-selector').value;
            if (!driveId) {
                document.getElementById('attendance-history').innerHTML = '<p class="text-muted">Select a drive to load attendance.</p>';
                return;
            }
            try {
                const response = await fetch(`/api/attendance/opportunity/${driveId}`);
                if (!response.ok) throw new Error('HTTP ' + response.status);
                const data = await response.json();

                if (!data.attendance || data.attendance.length === 0) {
                    document.getElementById('attendance-history').innerHTML = '<p class="text-muted">No attendance records for this drive yet.</p>';
                    return;
                }
                const items = data.attendance.slice(0, 20).map(r => `<tr><td>${r.student_name}</td><td>${r.roll_no || '-'}</td><td>${r.branch || '-'}</td><td>${new Date(r.scanned_at).toLocaleTimeString()}</td></tr>`).join('');
                document.getElementById('attendance-history').innerHTML = `<div class="table-responsive"><table class="table table-sm table-hover"><thead><tr><th>Student</th><th>Roll No</th><th>Branch</th><th>Time</th></tr></thead><tbody>${items}</tbody></table></div>`;
            } catch (err) {
                console.error('loadAttendanceHistory error', err);
                document.getElementById('attendance-history').innerHTML = '<p class="text-danger">Failed to load history.</p>';
            }
        }

        loadPlacementDrives();
    </script>
